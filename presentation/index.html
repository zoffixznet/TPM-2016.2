Rewrite To Be Shorter


 use Crypt::Bcrypt;
 my @passes = rand.Str xx 32;

 my @hashes;
 for @passes {
     @hashes.push: bcrypt-hash $^pass, :15rounds;
 }
 say @hashes;

 

Single-Threaded


 use Crypt::Bcrypt;
 my @passes = rand.Str xx 32;

 my @hashes = @passes.map: {bcrypt-hash $^pass, :15rounds};
 say @hashes;

Multi-Threaded


 use Crypt::Bcrypt;
 my @passes = rand.Str xx 32;

 my @hashes = @passes.race.map: {bcrypt-hash $^pass, :15rounds};
 say @hashes;
